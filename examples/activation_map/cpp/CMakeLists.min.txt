cmake_minimum_required(VERSION 3.20)
project(holoscan_activation_map CXX)

# Finds the package holoscan
find_package(holoscan REQUIRED CONFIG
             PATHS "/opt/nvidia/holoscan" "/workspace/holoscan-sdk/install")

add_executable(activation_map
  activation_map.cpp
)

target_link_libraries(activation_map
  PRIVATE
  holoscan::core
  holoscan::ops::inference
  CUDA::cuda_driver
)

# Testing
if(BUILD_TESTING)
  file(COPY ${CMAKE_CURRENT_SOURCE_DIR}/activation_map.yaml
      DESTINATION  ${CMAKE_CURRENT_BINARY_DIR}/)

  find_package(Python3 REQUIRED COMPONENTS Interpreter)

  add_test(NAME EXAMPLE_CPP_ACTIVATION_MAP_TEST
    COMMAND bash -c
      "${CMAKE_CURRENT_BINARY_DIR}/activation_map --config ${CMAKE_CURRENT_BINARY_DIR}/activation_map.yaml 2>&1 | \
       ${Python3_EXECUTABLE} ${CMAKE_SOURCE_DIR}/../bin/test_pattern_validation.py \
         --pass-pattern 'Model first, second, third was selected to infer' \
         --pass-pattern '7, 7, 7, 7, 7, 7, 7, 7, 7, 7, 7, 7, 7, 7, 7, 7' \
         --pass-pattern '8, 8, 8, 8, 8, 8, 8, 8, 8, 8, 8, 8, 8, 8, 8, 8' \
         --pass-pattern '9, 9, 9, 9, 9, 9, 9, 9, 9, 9, 9, 9, 9, 9, 9, 9'"
    WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}
  )
endif()
