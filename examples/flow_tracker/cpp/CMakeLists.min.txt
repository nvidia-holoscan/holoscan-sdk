cmake_minimum_required(VERSION 3.20)
project(holoscan_flow_tracker CXX)

# Finds the package holoscan
find_package(holoscan REQUIRED CONFIG
             PATHS "/opt/nvidia/holoscan" "/workspace/holoscan-sdk/install")

add_executable(flow_tracker
  flow_tracker.cpp
)

target_link_libraries(flow_tracker
  PRIVATE
  holoscan::core
)

# Testing
if(BUILD_TESTING)
  find_package(Python3 REQUIRED COMPONENTS Interpreter)

  add_test(NAME EXAMPLE_CPP_FLOW_TRACKER_TEST
    COMMAND bash -c
      "${CMAKE_CURRENT_BINARY_DIR}/flow_tracker 2>&1 | \
      ${Python3_EXECUTABLE} ${CMAKE_SOURCE_DIR}/../bin/test_pattern_validation.py \
        --pass-pattern 'Data Flow Tracking Results' \
        --pass-pattern 'Path 1: root1,middle1,leaf1' \
        --pass-pattern 'Path 2: root1,middle1,leaf2' \
        --pass-pattern 'Path 3: root2,middle2,leaf2'"
    WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}
  )
endif()
